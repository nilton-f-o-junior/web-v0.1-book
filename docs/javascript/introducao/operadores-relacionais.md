# Operadores Relacionais

`>`

```js
// maior
let a = 2, b = 3;
console.log(a > b);
```

`<`

```js
// menor
let a = 2, b = 3;
console.log(a < b);
```

`>=`

```js
// maior igual
let a = 3, b = 3;
console.log(a >= b);

```

`<=`

```js
// menor igual
let a = 2, b = 3;
console.log(a <= b)
```

`==`

```js
// igual
let a = 2, b = 3;
console.log(a == b)
```

`!=`

```js
// negação
let a = 2, b = 3;
console.log(a != b)
```

`===`

```js
// igual + igual
let a = 2;
let b = "2";
console.log(a == b); // true

let c = 2;
let d = "2";
console.log(c === d); // false
```
