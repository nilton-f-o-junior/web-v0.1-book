# Operadores Aritméticos + Precedência

`+`

```js
// soma
let a = 3;
let b = 9;

console.log(a + b);
```

`-`

```js
// subtracão
let a = 3;
let b = 9;

console.log(a - b);
```

`*`

```js
// multiplicacão
let a = 3;
let b = 9;

console.log(a * b);
```

`/`

```js
// divisão
let a = 3;
let b = 9;

console.log(a / b);
```

`%`

```js
// resto
let a = 3;
let b = 9;

console.log(a % b);
```

`()`

```js
// segue a precedência matemática
let a = 3;
let b = 9;
let c = 2;

console.log(a + b * c);
console.log((a * b) + c);
```
